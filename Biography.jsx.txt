          <Link to="/" className="inline-flex items-center gap-2 text-white/70 hover:text-white text-sm mb-8 transition-colors">
            {t.backHome}
          </Link>

          <div className="flex flex-col md:flex-row items-center md:items-end gap-8">

              <div className="absolute -inset-2 rounded-full bg-gradient-to-br from-primary/60 to-accent/40 blur-sm" />
              <img
                src={PROFILE_PHOTO}

            </div>

            {/* Text */}
            <div className={`text-center md:text-left ${fontClass}`}>

        </div>
      </div>

        {/* Vertical timeline line */}
        <div className="relative">
          <div className="hidden md:block absolute left-[27px] top-0 bottom-0 w-0.5 bg-gradient-to-b from-primary/40 via-border to-transparent" />

          <div className="space-y-10">
            {SECTIONS.map((sec, i) => (
              <div key={i} className="flex gap-6 items-start group">
                {/* Icon dot */}
                <div className={`relative z-10 w-14 h-14 rounded-2xl bg-gradient-to-br ${sec.color} border ${sec.border} flex items-center justify-center text-2xl flex-shrink-0 shadow-sm group-hover:scale-110 transition-transform duration-200`}>
                  {sec.icon}
                </div>

                {/* Card */}
                <div className="flex-1 bg-card border border-border rounded-2xl shadow-sm hover:shadow-lg transition-shadow duration-300 overflow-hidden">

                      {getTitle(sec, lang)}
                    </h2>
                  </div>
                  {/* Card body */}
                  <div className="px-6 py-5">
                    {getContent(sec, lang).split('\n\n').map((para, j) => (

                        {para}
                      </p>
                    ))}
                  </div>
                </div>
              </div>
            ))}
          </div>
        </div>

        {/* ── Closing Quote ── */}
        <div className="mt-16 relative overflow-hidden bg-gradient-to-br from-primary/8 via-card to-accent/8 border border-primary/25 rounded-3xl p-10 text-center shadow-sm">
          <div className="absolute top-0 left-0 w-32 h-32 bg-primary/5 rounded-full -translate-x-1/2 -translate-y-1/2 blur-2xl" />
          <div className="absolute bottom-0 right-0 w-32 h-32 bg-accent/5 rounded-full translate-x-1/2 translate-y-1/2 blur-2xl" />
          <p className="relative text-5xl mb-5">🙏</p>
          <p className={`relative font-serif text-2xl md:text-3xl font-semibold text-foreground italic mb-4 leading-snug ${fontClass}`}>
            {t.quote}
          </p>
          <div className="w-10 h-0.5 bg-primary mx-auto mb-3" />
          <p className="relative text-muted-foreground text-sm font-medium tracking-wide">
            — {lang === 'ne' ? 'सागर ग्यालन मगर' : lang === 'ko' ? '사가르 갈란 마가르' : 'Sagar Gyalan Magar'}
          </p>
        </div>

        {/* Back to home button */}
        <div className="mt-10 text-center">
          <Link
            to="/"
            className="inline-flex items-center gap-2 px-8 py-3 bg-primary text-primary-foreground rounded-full font-semibold hover:bg-primary/90 transition-all hover:scale-105 shadow-md"
          >
            {t.backHome}
          </Link>
        </div>
      </div>

      <Footer />
    </div>
  );
}
